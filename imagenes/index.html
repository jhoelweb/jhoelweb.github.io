PRESENT SIMPLE

ESTRUCTURE

"AFIRMATIVE"= SUJETO + VERB BASE( "S" EN 3PERSON) + COMPLEMENT

"NEGATIVE" = SUJETO + DO/DOES + NOT (DON'T DOESN'T) + VERB BASE + COMPLEMENT

"QUESTION" = DO/DOES + SUJETO + VERB BASE + COMPLEMTE + QUESTION MARK

"QUESTION - " = DON'T/DOESN'T + SUJETO + VERB BASE + COMPLEMENT + QUESTION MARK
